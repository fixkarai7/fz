from collections import Counter
import re

text = """
Cheap watches available now! Best cheap watches for you.
Buy cheap watches online. Cheap cheap cheap watches watches!
"""

words = re.findall(r'\b\w+\b', text.lower())
count = Counter(words)
SPAM_LIMIT = 4

print("Word counts:")
for w, c in count.items():
    print(w, ":", c)

print("\nSpammy Words:")
for w, c in count.items():
    if c >= SPAM_LIMIT:
        print(f"{w} appears {c} times (SPAM)")
